#include <stdio.h>
#include <sys/ipc.h>
#include <sys/msg.h>
#include <sys/types.h>
#include <unistd.h>
#include <string.h>

struct msg_buffer {
    long msg_type;
    char msg_text[100];
};

int main() {
    int msgid;
    key_t key = 1234;

    // Create message queue
    msgid = msgget(key, 0666 | IPC_CREAT);
    if (msgid == -1) {
        printf("Message queue creation failed\n");
        return 1;
    }

    int pid = fork();

    if (pid > 0) {
        // ---------------- Sender ----------------
        struct msg_buffer message;
        message.msg_type = 1;

        printf("Enter message to send: ");
        fgets(message.msg_text, sizeof(message.msg_text), stdin);

        msgsnd(msgid, &message, sizeof(message.msg_text), 0);
        printf("Parent (Sender): Message sent\n");
    }
    else {
        // ---------------- Receiver ----------------
        sleep(2); // wait for sender

        struct msg_buffer message;
        msgrcv(msgid, &message, sizeof(message.msg_text), 1, 0);

        printf("Child (Receiver): Message received = %s", message.msg_text);

        // delete queue
        msgctl(msgid, IPC_RMID, NULL);
    }

    return 0;
}

output:
Parent (Sender): Message sent
Child (Receiver): Message received = task is done!
